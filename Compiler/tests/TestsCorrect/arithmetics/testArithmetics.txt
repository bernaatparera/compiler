function main{

    //test that displays the correct use of arithmetics operations

    int number1 =1;
    int number2 = 1+1;

    int number3 = number1+number2;
    println(number3);

    number3 = -number1;
    println(number3);

    number3 = number1-number2;
    println(number3);

    number3 = -2*2;
    println(number3);

    number3 = number2 /2;
    println(number3);

    number3 = -number1+1;
    println(number3);

    //No parenthesis implemented, this is equivalent to
    //-number1 + 7 + (10*2)
    number3 = -number1+7+10*2;
    println(number3);
}
